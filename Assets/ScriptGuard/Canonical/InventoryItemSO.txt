    public void SpawnFromInventory(Vector3 spawnPosition)

        SpawnItem(spawnPosition, DragStack.stack.quantity);
    }

    public void SpawnItem(Vector3 position, int quantity)
    {
        GameObject droppedItem;

        if (itemPrefab != null)
        {
            // Instantiate mesh-based prefab
            droppedItem = Instantiate(itemPrefab, position, Quaternion.identity);
        }
        else
        {
            // Fallback to placeholder object with icon (not rendered in inventory)
            droppedItem = new GameObject(itemPrefab.name + "_Dropped");
            droppedItem.transform.position = position;

            // Optional: for debugging only — you could remove this if icon shouldn't show in world
            var renderer = droppedItem.AddComponent<SpriteRenderer>();
            renderer.sprite = itemImage;
            renderer.sortingOrder = 5;
        }

        // Attach pickup logic
        var pickup = droppedItem.GetComponent<InventoryItem>();
        pickup.Initiate(this, quantity);
    }
